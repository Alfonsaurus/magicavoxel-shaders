// v. 0.1.0
// xs noise [a] [b]

float rand(vec2 co){
  return fract(sin(dot(co.xy, vec2(12.9898, 78.233))) * 43758.5453);
}

float map(vec3 v) {
    float index = voxel(v);
    if (index == iColorIndex) {
        return iArgs[0] + (rand(v) * (iArgs[1] - iArgs[0]));
    }
    return index;
}
