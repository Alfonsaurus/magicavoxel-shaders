float c = max(1.0, iArgs[1]);

float map(vec3 v) {
    float x = voxel(v);
    if (x == iColorIndex) {
        for (float z = 1.0; z <= c; z += 1.0) {
            if (voxel(vec3(v.x, v.y, v.z + z)) > 0.0) {
                return x;
            }
        }
        return iArgs[0];
    }
    return x;
}
