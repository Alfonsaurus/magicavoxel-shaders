float map(vec3 v) {
	float offset = max(i_args[1], 1.0);
	float nz = (v.z + offset);

	if (v.z >= i_args[0]) {
		if (nz >= i_volume_size.z) {
			return 0.0;
		} else {
			return voxel(vec3(v.x, v.y, nz));
		}
	} else {
		return voxel(v);
	}
}
