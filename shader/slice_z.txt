float offset = max(i_args[1], 1.0);

float map(vec3 v) {
	float dz = v.z + offset;

	if (v.z >= i_args[0]) {
		if (dz >= i_volume_size.z) {
			return 0.0;
		} else {
			return voxel(vec3(v.x, v.y, dz));
		}
	} else {
		return voxel(v);
	}
}
