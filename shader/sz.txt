float map(vec3 v) {
	float offset = max(iArgs[1], 1.0);
	float nz = (v.z + offset);

	if (v.z >= iArgs[0]) {
		if (nz >= iVolumeSize.z) {
			return 0.0;
		} else {
			return voxel(vec3(v.x, v.y, nz));
		}
	} else {
		return voxel(v);
	}
}
