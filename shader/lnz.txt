float n = max(floor(iArgs[1]), 2.0);
float o = max(floor(iArgs[2]), 0.0);

float map(vec3 v) {
	float c = voxel(v);

	if (c == iColorIndex) {
		if (mod(v.z + o, n) == 0.0) {
			return iArgs[0];
		}
	}
	return c;
}
